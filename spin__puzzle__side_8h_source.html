679 bool SpinPuzzleSide<N, M>::rotate_internal_disk(double angle) {
680  if (!m_status.is_internal_disk_rotation_possible()) {
681  return false;
682  }
683  // normalize the angle
684  const double alpha =
685  std::fmod(m_status.get_central_disk_shift() + 360.0, 360.0);
686  const double new_shift_cdisk = std::fmod(alpha + angle + 360.0, 360.0);
687  // ======================================================================== //
688  // FROM BORDER ROTATION TO INTERNAL DISK ROTATION
689  if (m_status.get_trefoil_status(TIME::CURRENT) == TREFOIL::BORDER_ROTATION) {
690  if ((300 - TOLLERANCE_ANGLE <= alpha && alpha <= 300 + TOLLERANCE_ANGLE) ||
691  (60 - TOLLERANCE_ANGLE <= alpha && alpha <= 60 + TOLLERANCE_ANGLE)) {
692  prepare_from_border_rotation();
693  }
694  }
695 
696  angle = fmod(angle + 360.0, 360.0);
697  m_status.set_trefoil_status(TREFOIL::INVALID);
698 
699  if (120 - TOLLERANCE_ANGLE <= new_shift_cdisk &&
700  new_shift_cdisk < 240 - TOLLERANCE_ANGLE) {
701  // ROTATE marbles NORTH->EAST->WEST
702  auto north_it = this->begin(LEAF::NORTH) - 1;
703  auto east_it = this->begin(LEAF::EAST) - 1;
704  auto west_it = this->begin(LEAF::WEST) - 1;
705  for (int8_t i = 1; i <= 3; ++i, --north_it, --east_it, --west_it) {
706  std::iter_swap(north_it, west_it);
707  std::iter_swap(east_it, west_it);
708  }
709  // recalcualte the shift angle
710  const double updated_angle = new_shift_cdisk - 120;
711  m_status.set_central_disk_shift(updated_angle);
712  } else if (240 - TOLLERANCE_ANGLE <= new_shift_cdisk &&
713  new_shift_cdisk <= 240 + TOLLERANCE_ANGLE) {
714  // ROTATE marbles WEST->EAST->NORTH
715  auto north_it = this->begin(LEAF::NORTH) - 1;
716  auto east_it = this->begin(LEAF::EAST) - 1;
717  auto west_it = this->begin(LEAF::WEST) - 1;
718  for (int8_t i = 1; i <= 3; ++i, --north_it, --east_it, --west_it) {
719  std::iter_swap(north_it, east_it);
720  std::iter_swap(east_it, west_it);
721  }
722  // recalcualte the shift angle
723  const double updated_angle = new_shift_cdisk - 240;
724  m_status.set_central_disk_shift(updated_angle);
725  } else {
726  m_status.set_central_disk_shift(new_shift_cdisk);
727  }
728  if ((0 <= new_shift_cdisk && new_shift_cdisk <= TOLLERANCE_ANGLE) ||

733  new_shift_cdisk <= 240 + TOLLERANCE_ANGLE)) {
734  m_status.set_trefoil_status(TREFOIL::LEAF_ROTATION);
735  }
736
